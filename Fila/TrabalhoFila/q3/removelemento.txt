public T  RemoveElemento(T E){
    No<T> atual = cabeça;
    No<T> anterior = cabeça;
    int posição = 0;
    while(atual != null && atual.GetElemento() != E){
        anterior=atual;
        atual = atual.getProximo();
        posição++;
    }
    if(atual==null){
        return null;
    }else{
        if(cabeça == atual){
            cabeça=atual.getProximo();
        }
        else if(calda == atual){
            calda=anterior;
        }
    }
}
